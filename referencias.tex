%%%%%%%%%%%%%%%%
%Las referencias deben de ser escritas en un solo estilo bibliográfico para uniformizar su presentación
%en el documento, como el IEEE para Electrónica y el Vancouver para Biomédica.
%
% las referencias utilizadas pueden ser de varios tipos, solo se colocaron aquí los más utilizados.
% se coloca en \bibitem el nombre de referencia, para ser utilizado automaticamente en todo el documento
%
% en el documento se le llama a las referencias con el comando cite, ejemplo: \cite{libr1}
%
% cuando coloco la palabra autor siempre me refiero al formato:
% apellidoP-apellidoM iniciales de los nombres; ejemplo: Duarte-Villaseñor M.A.


%%%%%%%%%%%%%%%% formato IEEE (electrónica)

%libro
\bibitem{libr1}
autor1, autor2. nombre-del-libro. editorial, edición, lugar, año.

%capitulo de libro
\bibitem{capi1}
autor1, autor2, autor3. nombre-del-capitulo-del-libro, cap.X del nombre-del-libro. pp. X-X, nombre de los editores, editorial. edición, lugar, año.

%Tesis
\bibitem{tesis1}
autor. nombre-de-la-tesis. Tesis de Licenciatura-Maestria-Doctorado, Escuela-o-Facultad del Universidad-o-Instituto, lugar, año.

%pag web
\bibitem{pagi1}
autor1, autor2. titulo-pag [Online]. lugar, fecha del trabajo. Consultado el: día-mes-año.\\
http://www.pagina.com
%ejemplo:
\bibitem{webMCI}
Instituto Tecnológico de Tijuana pagina WEB oficial. Tijuana, B.C. Consultado el: 04 de abril de 2023.\\
https://www.tijuana.tecnm.mx/

%articulos de revista
\bibitem{revi1}
autor1, autor2, autor3. nombre-del-articulo, en nombre-de-la-revista. Vol.X, No.XX, mes-y-año, pp. yy-zz.

%articulos de conferencias
\bibitem{confe1}
autor1, autor2. nombre-de-la-conferencia, en nombre-del-evento. mes año, pp. yy-zz.


%%%%%%%%%%%%%%%% formato Vancouver (biomédica)
%libro
\bibitem{libr2}
autor1, autor2. nombre-del-libro. editorial, edición, lugar, año.

%articulos de revista
\bibitem{revi2}
autor1, autor2, autor3. nombre-del-articulo. nombre-de-la-revista. año; Vol(No.): pInicial-pFinal.

%pag web
\bibitem{pagi2}
nombre [Internet]. fecha del copyright o última actualización, si se desconoce, poner n.d.; [consultado fecha]. disponible en: http://www.pagina.com
